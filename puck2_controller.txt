from controller import Robot

def run_robot(robot):
    time_step = 32
    max_speed = 6.28

    # Motors
    left_motor = robot.getDevice('left wheel motor')
    right_motor = robot.getDevice('right wheel motor')
    
    left_motor.setPosition(float('inf'))
    right_motor.setPosition(float('inf'))
    
    left_motor.setVelocity(0.0)
    right_motor.setVelocity(0.0)

    # Enable IR sensors
    left_ir = robot.getDevice('ir1')
    right_ir = robot.getDevice('ir0')
    left_ir.enable(time_step)
    right_ir.enable(time_step)
    
    while robot.step(time_step) != -1:
        left_ir_value = left_ir.getValue()
        right_ir_value = right_ir.getValue()
        
        print("Left IR: {:.2f}, Right IR: {:.2f}".format(left_ir_value, right_ir_value))

        # Default speeds
        left_speed = right_speed = 0.25 * max_speed

        # Based on your custom thresholds
        if (left_ir_value > right_ir_value) and (4 < left_ir_value < 15):
            print("Turning left")
            left_speed = -0.25 * max_speed
            right_speed = 0.25 * max_speed
            
        elif (right_ir_value > left_ir_value) and (4 < right_ir_value < 15):
            print("Turning right")
            left_speed = 0.25 * max_speed
            right_speed = -0.25 * max_speed
            
        # elif (1 < left_ir_value < 15) and (1 < right_ir_value < 15):
            # print("On the line - moving forward")
            # left_speed = right_speed = 0.25 * max_speed
            
        # else:
            # print("Line lost - stopping")
            # left_speed = right_speed = 0.0

        # Apply speeds
        left_motor.setVelocity(left_speed)
        right_motor.setVelocity(right_speed)

if __name__ == "__main__":
    my_robot = Robot()
    run_robot(my_robot)
